import requests

def lookup_number(phone_number):
    api_key = "ForApp&adhartofam"
    url = "https://danger-vip-key.shop/api.php"
    params = {
        "key": api_key,
        "number": phone_number
    }
    
    response = requests.get(url, params=params)
    return response.json()

# Usage
